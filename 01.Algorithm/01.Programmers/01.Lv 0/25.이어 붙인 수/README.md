class Solution {
    public int solution(int[] num_list) {
        int answer = 0;
        
        int sum = 0;

        String sumstr2 = "";    
        
        for(int i = 0; i < num_list.length; i++) {
            
            if(num_list[i] % 2 == 1) {
                sumstr += "" + num_list[i];
            }
            
            if(num_list[i] % 2 == 0) {
                sumstr2 += "" + num_list[i];
            
            if(num_list.length -1 == i ) {
            }
                int intstr = Integer.parseInt(sumstr);
                int intstr2 = Integer.parseInt(sumstr2);

            }

        }

    }
}
